# @holeauth/plugin-rbac

Add Role-Based Access Control (RBAC) with wildcard permissions, hierarchical groups, and runtime reload.

## Quick Example

```ts title="lib/auth.ts"
import { defineHoleauth } from '@holeauth/core';
import { rbac } from '@holeauth/plugin-rbac';
import { createRbacAdapter } from '@holeauth/rbac-drizzle/pg';
import { loadRbacYaml } from '@holeauth/rbac-yaml';
import { db } from '@/db';
import { tables } from '@/db/schema';

const yaml = loadRbacYaml('./holeauth.rbac.yml', {
  watch: process.env.NODE_ENV !== 'production',
});

const plugin = rbac({
  groups: yaml.snapshot.groups,
  adapter: createRbacAdapter({ db, tables }),
});

yaml.onReload((snapshot) => plugin.reload(snapshot.groups));

export const auth = defineHoleauth({
  adapter: drizzleAdapter(db),
  secrets: { jwtSecret: process.env.HOLEAUTH_SECRET! },
  plugins: [plugin],
});
```

```ts title="lib/auth.ts"
import { rbac } from '@holeauth/plugin-rbac';
import { createRbacAdapter } from '@holeauth/rbac-drizzle/pg';
import { createRedisCache } from '@/lib/rbac-cache';
import { db } from '@/db';
import { tables } from '@/db/schema';

const plugin = rbac({
  groups: [
    {
      id: 'user',
      default: true,
      effective: ['posts.read', 'comments.create'],
    },
    {
      id: 'admin',
      effective: ['*'],
    },
  ],
  adapter: createRbacAdapter({ db, tables }),
  cacheTtlMs: 60000,
  cache: createRedisCache(),
});
```

## RbacAdapter Interface

If you are not using [@holeauth/rbac-drizzle](drizzle), implement this interface to provide custom storage.

```ts
interface RbacAdapter {
  listUserGroups(userId: string): Promise<string[]>;
  assignGroup(userId: string, groupId: string): Promise<void>;
  removeGroup(userId: string, groupId: string): Promise<void>;
  listUserPermissions(userId: string): Promise<string[]>;
  grantPermission(userId: string, node: string): Promise<void>;
  revokePermission(userId: string, node: string): Promise<void>;
  listAllGroupAssignments(): Promise<UserGroupAssignment[]>;
  purgeUser(userId: string): Promise<void>;
}
```

## API Surface

The plugin exposes the following methods on your auth instance at `auth.rbac.*`:

### can()

Checks if a user has permission for a single node.

```ts
auth.rbac.can(userId: string, node: string): Promise<boolean>
```

**Example**

```ts title="app/api/posts/route.ts"
import { auth } from '@/lib/auth';

export async function POST(req: Request) {
  const session = await auth.validateCurrentRequest(req);
  if (!session) return new Response('Unauthorized', { status: 401 });

  const allowed = await auth.rbac.can(session.userId, 'posts.create');
  if (!allowed) return new Response('Forbidden', { status: 403 });

  // Create post...
}
```

### canAll()

Checks if a user has permission for all nodes in an array.

```ts
auth.rbac.canAll(userId: string, nodes: string[]): Promise<boolean>
```

### canAny()

Checks if a user has permission for at least one node in an array.

```ts
auth.rbac.canAny(userId: string, nodes: string[]): Promise<boolean>
```

### listGroups()

Returns all defined groups.

```ts
auth.rbac.listGroups(): GroupDefinition[]
```

### getGroup()

Returns a single group definition by ID.

```ts
auth.rbac.getGroup(id: string): GroupDefinition | null
```

### getUserGroups()

Returns the groups assigned to a user (includes the default group).

```ts
auth.rbac.getUserGroups(userId: string): Promise<GroupDefinition[]>
```

### getUserPermissions()

Returns the effective permission nodes for a user (cached).

```ts
auth.rbac.getUserPermissions(userId: string): Promise<string[]>
```

### getEffectiveNodes()

Alias for `getUserPermissions`.

```ts
auth.rbac.getEffectiveNodes(userId: string): Promise<string[]>
```

### assignGroup()

Assigns a group to a user.

```ts
auth.rbac.assignGroup(userId: string, groupId: string): Promise<void>
```

### removeGroup()

Removes a group assignment from a user.

```ts
auth.rbac.removeGroup(userId: string, groupId: string): Promise<void>
```

### grant()

Grants a direct permission to a user (bypassing groups).

```ts
auth.rbac.grant(userId: string, node: string): Promise<void>
```

### revoke()

Revokes a direct permission from a user.

```ts
auth.rbac.revoke(userId: string, node: string): Promise<void>
```

### reload()

Replaces the group set at runtime (used for YAML hot-reload).

```ts
auth.rbac.reload(groups: GroupDefinition[]): void
```

**Example**

```ts title="lib/auth.ts"
import { loadRbacYaml } from '@holeauth/rbac-yaml';

const yaml = loadRbacYaml('./holeauth.rbac.yml', { watch: true });
const plugin = rbac({ groups: yaml.snapshot.groups, adapter });

yaml.onReload((snapshot) => plugin.reload(snapshot.groups));
```

### listOrphans()

Returns all user-group assignments that reference a group ID no longer defined in the current snapshot.

```ts
auth.rbac.listOrphans(): Promise<UserGroupAssignment[]>
```

### snapshot()

Returns the current group set and default group ID.

```ts
auth.rbac.snapshot(): { groups: GroupDefinition[]; defaultGroupId: string }
```

## Permission Node Syntax

Permission nodes are dot-separated strings with optional wildcard matching:

- `posts.create` — Exact match
- `posts.*` — Matches `posts.create`, `posts.delete`, `posts.update`, etc.
- `*` — Matches everything
- `!posts.delete` — Negation (revokes a permission)

**Priority**: Negations always win. Within a user's effective permission list, the last matching pattern determines the result. Groups are ordered by `priority` (lower = higher precedence).

**Example**

```yaml
groups:
  admin:
    default: false
    priority: 0
    permissions:
      - '*'
  editor:
    default: false
    priority: 1
    permissions:
      - 'posts.*'
      - '!posts.delete'
  user:
    default: true
    priority: 2
    permissions:
      - 'posts.read'
      - 'comments.create'
```

## Utility Exports

### matchNodes()

Checks if a permission node matches a user's effective permissions (pure function, no database).

```ts
function matchNodes(effective: string[], query: string): boolean
```

### matchNodesAll()

```ts
function matchNodesAll(effective: string[], queries: string[]): boolean
```

### matchNodesAny()

```ts
function matchNodesAny(effective: string[], queries: string[]): boolean
```

### matchPattern()

Checks if a wildcard pattern matches a concrete permission node.

```ts
function matchPattern(pattern: string, node: string): boolean
```

**Example**

```ts
import { matchPattern } from '@holeauth/plugin-rbac';

matchPattern('posts.*', 'posts.create'); // true
matchPattern('posts.*', 'comments.create'); // false
matchPattern('*', 'anything'); // true
```

### Custom Permission Matching

`matchNodes` is the same function used internally by `can()`. Use it to evaluate permission decisions in pure code paths — middleware, schema layers, server actions — without round-tripping through the plugin API:

```ts
import { matchNodes } from '@holeauth/plugin-rbac';

const effective = ['posts.*', '!posts.delete', 'comments.create'];

matchNodes(effective, 'posts.create');     // true
matchNodes(effective, 'posts.delete');     // false (negation wins)
matchNodes(effective, 'comments.create');  // true
```

Pair it with `auth.rbac.getEffectiveNodes(userId)` (also exposed by `useRbacSnapshot()` on the client) to perform offline checks against a cached snapshot — for example, to filter a navigation menu without an extra request per item.

## Cache Adapter

Per-user permission lookups go through a `RbacCacheAdapter`. The default implementation is an in-memory TTL map; pass a custom adapter via `rbac({ cache, ... })` to share the cache across multiple instances.

```ts
interface RbacCacheAdapter {
  get(userId: string): string[] | undefined | Promise<string[] | undefined>;
  set(userId: string, nodes: string[]): void | Promise<void>;
  invalidate(userId: string): void | Promise<void>;
  clear(): void | Promise<void>;
}
```

    },
    set: {
      description: 'Store the effective nodes for a user. The plugin calls `set` after every cache miss.',

    invalidate: {

    },
    clear: {
      description: 'Drop every entry. Called by `reload()` so that group-definition changes take effect immediately.',
      type: '() => void | Promise<void>',
    },
  }}

### defaultRbacCache()

Build the in-memory TTL cache that ships with the plugin. Exposed as an export so you can wrap it (e.g. to add metrics).

```ts
function defaultRbacCache(ttlMs: number): RbacCacheAdapter
```

### Redis-backed cache

For multi-instance deployments, back the cache with Redis so that an `invalidate(userId)` call on one node clears the cached snapshot everywhere. The example below uses `ioredis`, but any client with `get` / `set` / `del` works.

```ts title="lib/rbac-cache.ts"
import type { RbacCacheAdapter } from '@holeauth/plugin-rbac';
import { redis } from './redis';

export function redisRbacCache(ttlSeconds = 60): RbacCacheAdapter {
  const key = (userId: string) => `rbac:nodes:${userId}`;
  return {
    async get(userId) {
      const raw = await redis.get(key(userId));
      return raw ? (JSON.parse(raw) as string[]) : undefined;
    },
    async set(userId, nodes) {
      await redis.set(key(userId), JSON.stringify(nodes), 'EX', ttlSeconds);
    },
    async invalidate(userId) {
      await redis.del(key(userId));
    },
    async clear() {
      // Optional — replace with a SCAN over the `rbac:nodes:*` prefix in production.
    },
  };
}
```

```ts title="lib/auth.ts"
import { rbac } from '@holeauth/plugin-rbac';
import { redisRbacCache } from './rbac-cache';

const plugin = rbac({
  groups,
  adapter: rbacAdapter,
  cache: redisRbacCache(60),
});
```

<Callout type="warn">
  `clear()` runs on every `reload()` call. A no-op implementation is acceptable when TTLs are short; otherwise scan the key prefix and delete to avoid serving stale snapshots after a group-definition change.
</Callout>
