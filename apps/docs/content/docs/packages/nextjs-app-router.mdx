# @holeauth/nextjs-app-router

Next.js App Router bindings for holeauth. Provides a route handler for all auth endpoints, middleware for session management, and server helpers for React Server Components.

## Quick example

```ts title="lib/auth.ts"
import { createAuthHandler } from '@holeauth/nextjs-app-router';
import { drizzleAdapter } from '@holeauth/adapter-drizzle/pg';
import { db } from './db';

export const auth = createAuthHandler({
  adapter: drizzleAdapter(db),
  secrets: {
    jwtSecret: process.env.HOLEAUTH_SECRET!,
  },
});
```

```ts title="app/api/auth/[...holeauth]/route.ts"
import { auth } from '@/lib/auth';

export const { GET, POST } = auth.handlers;
```

## createAuthHandler

Creates a Next.js App Router-compatible auth instance with GET and POST handlers.

```ts
function createAuthHandler<
  const Plugins extends readonly HoleauthPlugin<string, unknown>[] = [],
>(
  config: HoleauthConfig,
  opts?: DispatchOptions,
): NextHoleauth<Plugins>;
```

<TypeTable
  type={{
    config: {
      description: 'Holeauth configuration (adapter, secrets, session, plugins).',
      type: 'HoleauthConfig',
      required: true,
    },
    opts: {
      description: 'Optional dispatcher options (custom route overrides).',
      type: 'DispatchOptions',
    },
  }}
/>

Returns a `NextHoleauth` instance with:

- `handlers.GET` — GET route handler for `/api/auth/*`
- `handlers.POST` — POST route handler for `/api/auth/*`
- `getSession(accessToken?)` — Validate an access token or read from `cookies()` in an RSC
- All methods from `HoleauthInstance` (flows, jwt, session, password, otp, sso, etc.)
- Plugin APIs when plugins are registered

```ts title="app/api/auth/[...holeauth]/route.ts"
import { auth } from '@/lib/auth';

export const { GET, POST } = auth.handlers;
```

## getSession

Server helper: validate an access token or read the current session from Next.js cookies.

```ts
async function getSession(accessToken?: string): Promise<SessionData | null>;
```

When called without an `accessToken` argument, reads the access token from `cookies()` in a React Server Component.

```ts title="app/dashboard/page.tsx"
import { auth } from '@/lib/auth';
import { redirect } from 'next/navigation';

export default async function DashboardPage() {
  const session = await auth.getSession();
  if (!session) {
    redirect('/login');
  }

  return <div>Welcome, {session.userId}</div>;
}
```

## Middleware

Middleware export for session validation and refresh.

```ts title="middleware.ts"
import { createMiddleware } from '@holeauth/nextjs-app-router/middleware';
import { auth } from '@/lib/auth';

export default createMiddleware(auth, {
  protectedPaths: ['/dashboard', '/settings'],
  publicPaths: ['/login', '/register'],
  redirectTo: '/login',
});

export const config = {
  matcher: ['/((?!_next/static|_next/image|favicon.ico).*)'],
};
```

## validateCurrentRequest

Low-level helper: validate the current request and return the session.

```ts
async function validateCurrentRequest(
  auth: HoleauthInstance,
): Promise<SessionData | null>;
```

Reads cookies from the Next.js `cookies()` API and validates the access token.

```ts title="app/api/protected/route.ts"
import { validateCurrentRequest } from '@holeauth/nextjs-app-router';
import { auth } from '@/lib/auth';
import { NextResponse } from 'next/server';

export async function GET() {
  const session = await validateCurrentRequest(auth);
  if (!session) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  return NextResponse.json({ userId: session.userId });
}
```

## getFullSession

Server helper: validate the current request and return the session with user data and RBAC snapshot.

```ts
async function getFullSession(
  auth: HoleauthInstance,
): Promise<{
  session: SessionData;
  user: AdapterUser;
  permissions?: string[];
  groups?: string[];
} | null>;
```

<Callout type="info">
  This helper automatically fetches the user record and RBAC snapshot (if `plugin-rbac` is registered) in a single database query. Use it when you need user data or permissions in a Server Component.
</Callout>

```ts title="app/admin/page.tsx"
import { getFullSession } from '@holeauth/nextjs-app-router';
import { auth } from '@/lib/auth';
import { redirect } from 'next/navigation';

export default async function AdminPage() {
  const result = await getFullSession(auth);
  if (!result) {
    redirect('/login');
  }

  if (!result.permissions?.includes('admin.access')) {
    redirect('/');
  }

  return <div>Welcome, {result.user.email}</div>;
}
```

## DispatchOptions

`createAuthHandler` accepts a second argument that tunes how the App Router route handler dispatches incoming requests. The same options object is re-exported from `@holeauth/nextjs-app-router` for use in custom integrations.

```ts
interface DispatchOptions {
  basePath?: string;
  defaultRedirect?: string;
}
```

<TypeTable
  type={{
    basePath: {
      description: 'Mount prefix stripped from the URL before dispatching. Must match the location of the catch-all route file.',
      type: 'string',
      default: '"/api/auth"',
    },
    defaultRedirect: {
      description: 'Destination used after a successful SSO callback when no explicit `redirect` query parameter is supplied.',
      type: 'string',
    },
  }}
/>

```ts title="app/api/auth/[...holeauth]/route.ts"
import { createAuthHandler } from '@holeauth/nextjs-app-router';
import { auth } from '@/lib/auth';

const handlers = createAuthHandler(auth, {
  basePath: '/api/auth',
  defaultRedirect: '/dashboard',
});

export const GET = handlers.GET;
export const POST = handlers.POST;
```

<Callout type="warn">
  Changing `basePath` requires updating both the file path of the catch-all route and the `basePath` field on `defineHoleauth` — the cookie domain and CSRF binding are derived from the same value.
</Callout>

## Middleware Patterns

The middleware helper composes with any other Next.js middleware. The patterns below cover the most common production setups.

### Restricting to authenticated paths

```ts title="middleware.ts"
import { createMiddleware } from '@holeauth/nextjs-app-router/middleware';
import { auth } from '@/lib/auth';

export default createMiddleware(auth, {
  protectedPaths: ['/dashboard', '/settings'],
  publicPaths: ['/', '/login', '/register'],
  redirectTo: '/login',
});

export const config = {
  // Exclude static assets and the auth API itself so the middleware never
  // intercepts its own refresh roundtrip.
  matcher: ['/((?!_next/static|_next/image|favicon.ico|api/auth).*)'],
};
```

### Combining with RBAC

The middleware only validates the session; route-level permission checks belong in the page or route handler. Use `getFullSession` in a Server Component to short-circuit a request when a permission is missing:

```ts title="app/admin/layout.tsx"
import { redirect } from 'next/navigation';
import { getFullSession } from '@holeauth/nextjs-app-router';
import { auth } from '@/lib/auth';

export default async function AdminLayout({ children }: { children: React.ReactNode }) {
  const result = await getFullSession(auth);
  if (!result) redirect('/login');
  if (!result.permissions?.includes('admin.access')) redirect('/');
  return <>{children}</>;
}
```

### Chaining with other middleware

Wrap the holeauth middleware to add i18n redirection, feature flags, or A/B test cookies before authentication runs.

```ts title="middleware.ts"
import { NextRequest, NextResponse } from 'next/server';
import { createMiddleware } from '@holeauth/nextjs-app-router/middleware';
import { auth } from '@/lib/auth';

const authMiddleware = createMiddleware(auth, {
  protectedPaths: ['/dashboard'],
  redirectTo: '/login',
});

export default async function middleware(req: NextRequest) {
  // Locale rewrite first — keeps cookies on the right URL.
  const locale = req.cookies.get('locale')?.value ?? 'en';
  if (!req.nextUrl.pathname.startsWith(`/${locale}`)) {
    return NextResponse.redirect(new URL(`/${locale}${req.nextUrl.pathname}`, req.url));
  }
  return authMiddleware(req);
}

export const config = {
  matcher: ['/((?!_next/static|_next/image|favicon.ico|api/auth).*)'],
};
```

<Callout type="info">
  Keep `api/auth` excluded from the matcher. The auth route handler manages its own cookies; running middleware on top of it can drop `Set-Cookie` headers during refresh rotation.
</Callout>
