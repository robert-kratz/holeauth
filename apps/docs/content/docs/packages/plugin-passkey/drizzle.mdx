# @holeauth/passkey-drizzle

Drizzle ORM adapter for `@holeauth/plugin-passkey` supporting PostgreSQL, MySQL, and SQLite.

## Subpath Exports

This package provides dialect-specific entry points:

- `@holeauth/passkey-drizzle/pg` — PostgreSQL
- `@holeauth/passkey-drizzle/mysql` — MySQL
- `@holeauth/passkey-drizzle/sqlite` — SQLite

All three expose the same API: `createPasskeyTables` and `createPasskeyAdapter`.

## Usage by Dialect

<Tabs items={['PostgreSQL', 'MySQL', 'SQLite']}>
  <Tab value="PostgreSQL">
    ```ts title="db/schema.ts"
    import { pgTable, text } from 'drizzle-orm/pg-core';
    import { createHoleauthTables } from '@holeauth/adapter-drizzle/pg';
    import { createPasskeyTables } from '@holeauth/passkey-drizzle/pg';

    export const users = pgTable('users', {
      id: text('id').primaryKey(),
      email: text('email').notNull().unique(),
      name: text('name'),
      image: text('image'),
    });

    const holeauth = createHoleauthTables({ usersTable: users });
    const passkeys = createPasskeyTables({ usersTable: users });

    export const tables = {
      ...holeauth.tables,
      ...passkeys.tables,
    };

    export const relations = {
      ...holeauth.relations,
      ...passkeys.relations,
    };
    ```

    ```ts title="lib/auth.ts"
    import { defineHoleauth } from '@holeauth/core';
    import { passkey } from '@holeauth/plugin-passkey';
    import { createPasskeyAdapter } from '@holeauth/passkey-drizzle/pg';
    import { db } from '@/db';
    import { tables } from '@/db/schema';

    export const auth = defineHoleauth({
      adapter: drizzleAdapter(db),
      secrets: { jwtSecret: process.env.HOLEAUTH_SECRET! },
      plugins: [
        passkey({
          adapter: createPasskeyAdapter({ db, tables }),
          rpID: 'example.com',
          rpOrigin: 'https://example.com',
        }),
      ],
    });
    ```

## API Reference

### createPasskeyTables()

Creates Drizzle table definitions for passkey credentials.

```ts
function createPasskeyTables<U extends UsersTable>(
  options: CreatePasskeyTablesOptions<U>
): { tables: { passkeys: ... }; relations: { passkeysRelations: ... } }
```

<TypeTable
  type={{
    usersTable: {
      description: 'Your Drizzle users table (must have an `id` column).',
      type: 'DrizzleTable',
      required: true,
    },
    prefix: {
      description: 'Prefix for the generated table name.',
      type: 'string',
      default: '"holeauth_passkey_"',
    },
  }}
/>

**Returns**

An object with:

- `tables.passkeys` — Drizzle table with columns: `id` (PK), `userId` (FK cascade), `credentialId` (unique), `publicKey`, `counter`, `transports`, `deviceName`, `createdAt`.
- `relations.passkeysRelations` — Drizzle relation joining `passkeys.userId` to `usersTable.id`.

**Example**

```ts title="db/schema.ts"
import { createPasskeyTables } from '@holeauth/passkey-drizzle/pg';
import { users } from './users';

const passkeys = createPasskeyTables({
  usersTable: users,
  prefix: 'app_passkey_',
});

export const tables = { ...passkeys.tables };
export const relations = { ...passkeys.relations };
```

### createPasskeyAdapter()

Constructs a `PasskeyAdapter` that implements the plugin's storage interface.

```ts
function createPasskeyAdapter(
  options: CreatePasskeyAdapterOptions
): PasskeyAdapter
```

<TypeTable
  type={{
    db: {
      description: 'Drizzle database instance.',
      type: 'DrizzleDB',
      required: true,
    },
    tables: {
      description: 'Tables object returned by `createPasskeyTables`.',
      type: 'Tables',
      required: true,
    },
    generateId: {
      description: 'Function that generates unique IDs for new credentials.',
      type: '() => string',
      default: 'crypto.randomUUID',
    },
  }}
/>

**Example**

```ts title="lib/auth.ts"
import { createPasskeyAdapter } from '@holeauth/passkey-drizzle/pg';
import { db } from '@/db';
import { tables } from '@/db/schema';

const adapter = createPasskeyAdapter({ db, tables });
```

## Schema Details

The `passkeys` table has the following columns (PostgreSQL example):

| Column | Type | Constraints |
| --- | --- | --- |
| `id` | `text` | Primary key |
| `userId` | `text` | Not null, foreign key → `users.id` (on delete cascade), indexed |
| `credentialId` | `text` | Not null, unique |
| `publicKey` | `text` | Not null |
| `counter` | `integer` | Not null, default `0` |
| `transports` | `text[]` | Nullable |
| `deviceName` | `text` | Nullable |
| `createdAt` | `timestamp` | Not null, default now |

The `counter` column is updated on every successful assertion to prevent replay attacks. A counter regression indicates a cloned authenticator and triggers a verification failure.

## Migration

After defining your schema, generate and run a migration:

```bash
npx drizzle-kit generate
npx drizzle-kit migrate
```
