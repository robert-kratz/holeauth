# Magic link and email OTP

Add passwordless sign-in to a Next.js Pages Router app. Users enter an email and receive either a one-click magic link or a six-digit OTP code.

<Callout type="info">
  This guide assumes you have completed the [core Next.js Pages Router setup](/getting-started/nextjs-pages-router).
</Callout>

<Step>
### Wire the plugin into your auth instance

```ts title="lib/auth.ts" {5,6,11,30-44}
import { createPagesAuthHandler } from '@holeauth/nextjs-pages-router';
import { createHoleauthAdapters } from '@holeauth/adapter-drizzle/pg';
import { db } from '@/db/client';
import { holeauth, magicLinkSchema } from '@/db/schema';
import { magicLink } from '@holeauth/plugin-magic-link';
import { createMagicLinkAdapter } from '@holeauth/magic-link-drizzle/pg';
import { sendTransactionalEmail } from '@/lib/email';

const AUTH_BASE_PATH = '/api/auth';
const adapters = createHoleauthAdapters({ db, tables: holeauth.tables });
const magicLinkAdapter = createMagicLinkAdapter({ db, tables: magicLinkSchema.tables });

export const auth = createPagesAuthHandler({
  secrets: { jwtSecret: process.env.HOLEAUTH_SECRET! },
  adapters: {
    user: adapters.user,
    session: adapters.session,
    account: adapters.account,
    auditLog: adapters.auditLog,
    verificationToken: adapters.verificationToken,
    transaction: adapters.transaction,
  },
  tokens: {
    accessTtl: 60 * 15,
    refreshTtl: 60 * 60 * 24 * 30,
    cookiePrefix: 'holeauth',
    sameSite: 'lax',
  },
  plugins: [
    magicLink({
      adapter: magicLinkAdapter,
      baseUrl: `${process.env.APP_URL}${AUTH_BASE_PATH}`,
      mode: 'both',
      role: 'primary',
      tokenTtlSeconds: 600,
      otpLength: 6,
      successRedirect: '/dashboard',
      errorRedirect: '/magic-link',
      expiredRedirect: '/magic-link/expired',
      sendEmail: async ({ email, url, code, type }) => {
        if (type === 'magic-link') await sendTransactionalEmail(email, 'Sign in', `Click: ${url}`);
        else await sendTransactionalEmail(email, 'Your code', `Code: ${code}`);
      },
    }),
  ] as const,
});
```

The plugin registers three routes under your existing `[...holeauth].ts` catch-all handler:

- `POST /api/auth/magic-link/request`
- `GET /api/auth/magic-link/consume?token=…`
- `POST /api/auth/magic-link/verify-otp`
</Step>

<Step>
### Build the request and verify pages

```tsx title="pages/magic-link/index.tsx"
import { useState, FormEvent } from 'react';

export default function MagicLinkPage() {
  const [email, setEmail] = useState('');
  const [sent, setSent] = useState(false);

  async function handleSubmit(e: FormEvent) {
    e.preventDefault();
    const res = await fetch('/api/auth/magic-link/request', {
      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({ email, type: 'magic-link' }),
    });
    if (res.ok) setSent(true);
  }

  if (sent) return <p>Check your inbox.</p>;
  return (
    <form onSubmit={handleSubmit}>
      <input type="email" value={email} onChange={(e) => setEmail(e.target.value)} required />
      <button type="submit">Send link</button>
    </form>
  );
}
```

```tsx title="pages/magic-link/verify.tsx"
import { useState, FormEvent } from 'react';
import { useRouter } from 'next/router';

export default function VerifyOtpPage() {
  const router = useRouter();
  const [email, setEmail] = useState('');
  const [code, setCode] = useState('');

  async function handleSubmit(e: FormEvent) {
    e.preventDefault();
    const res = await fetch('/api/auth/magic-link/verify-otp', {
      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({ email, code }),
    });
    if (res.ok) router.push('/dashboard');
  }

  return (
    <form onSubmit={handleSubmit}>
      <input type="email" placeholder="email" value={email} onChange={(e) => setEmail(e.target.value)} required />
      <input inputMode="numeric" pattern="\d*" placeholder="6-digit code" value={code} onChange={(e) => setCode(e.target.value)} required />
      <button type="submit">Verify</button>
    </form>
  );
}
```
</Step>

## Security notes

<Callout type="warn">
  Only the **hash** of each token / OTP is persisted. Use `https://` in production. Swap the default in-memory rate limiter for a Redis-backed one when scaling.
</Callout>
