# Next.js Pages Router

This guide walks you through integrating holeauth into a Next.js Pages Router project with Drizzle ORM. By the end, you will have email and password authentication, session management with automatic refresh-token rotation, and CSRF-protected endpoints.

<Step>
### Create database schema

Holeauth provides a table factory for each Drizzle dialect. Import the factory that matches your database, pass your existing users table, and export the resulting tables for use in migrations.

<Tabs items={['Postgres', 'MySQL', 'SQLite', 'Headless']}>
  <Tab value="Postgres">
```ts title="db/schema.ts"
import { pgTable, text, timestamp } from 'drizzle-orm/pg-core';
import { createHoleauthTables } from '@holeauth/adapter-drizzle/pg';

export const users = pgTable('user', {
  id: text('id').primaryKey().$defaultFn(() => crypto.randomUUID()),
  email: text('email').notNull().unique(),
  emailVerified: timestamp('email_verified', { withTimezone: true, mode: 'date' }),
  passwordHash: text('password_hash'),
  name: text('name'),
  image: text('image'),
  createdAt: timestamp('created_at', { withTimezone: true, mode: 'date' }).notNull().defaultNow(),
});

export const holeauth = createHoleauthTables({ usersTable: users });

// Re-export individual tables so drizzle-kit includes them in migrations.
// drizzle-kit only detects top-level named exports — it will not recurse
// into holeauth.tables automatically.
export const { sessions, accounts, verificationTokens, auditLog } = holeauth.tables;
```
</Tab>


<Callout type="warn">
  `createHoleauthTables()` returns a plain object — drizzle-kit will **not** detect tables nested
  inside it. The re-exports above are required so drizzle-kit includes all holeauth tables
  (including `holeauth_audit_log`) in your migration. Skipping them causes
  `relation "holeauth_audit_log" does not exist` errors at runtime.
</Callout>

<Step>
### Create the auth instance

<Tabs items={['Postgres', 'MySQL', 'SQLite', 'Headless']}>
  <Tab value="Postgres">
```ts title="lib/auth.ts"
import { createPagesAuthHandler } from '@holeauth/nextjs-pages-router';
import { createHoleauthAdapters } from '@holeauth/adapter-drizzle/pg';
import { db } from '@/db/client';
import { holeauth } from '@/db/schema';

const adapters = createHoleauthAdapters({ db, tables: holeauth.tables });

export const auth = createPagesAuthHandler({
  secrets: {
    jwtSecret: process.env.HOLEAUTH_SECRET!,
  },
  adapters: {
    user: adapters.user,
    session: adapters.session,
    account: adapters.account,
    auditLog: adapters.auditLog,
    verificationToken: adapters.verificationToken,
    transaction: adapters.transaction,
  },
  tokens: {
    accessTtl: 60 * 15, // 15 minutes
    refreshTtl: 60 * 60 * 24 * 30, // 30 days
    cookiePrefix: 'holeauth',
    sameSite: 'lax',
  },
  registration: {
    selfServe: true,
    inviteTtlSeconds: 7 * 24 * 60 * 60, // 7 days
    inviteUrl: ({ token }) =>
      `${process.env.APP_URL}/register/accept?token=${encodeURIComponent(token)}`,
  },
  allowDangerousEmailAccountLinking: false,
});
```

<Step>
### Mount the API route handler

Create a catch-all API route to serve all auth endpoints:

```ts title="pages/api/auth/[...holeauth].ts"
import { auth } from '@/lib/auth';

export default auth.handler;
```

This exposes the following routes:

- `POST /api/auth/register`
- `POST /api/auth/login`
- `POST /api/auth/logout`
- `POST /api/auth/refresh`
- `GET /api/auth/session`
- `POST /api/auth/change-password`
- `POST /api/auth/request-password-reset`
- `POST /api/auth/consume-password-reset`

See [Core API](/packages/core) for full request and response schemas.
</Step>

<Step>
### Add middleware (optional)

If you want to protect routes server-side or perform automatic token refresh, add middleware. Middleware works the same way in Pages Router as in App Router:

```ts title="middleware.ts"
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function middleware(request: NextRequest) {
  // Example: redirect unauthenticated users away from /dashboard
  const accessToken = request.cookies.get('holeauth.at')?.value;
  if (!accessToken && request.nextUrl.pathname.startsWith('/dashboard')) {
    return NextResponse.redirect(new URL('/login', request.url));
  }
  return NextResponse.next();
}

export const config = {
  matcher: ['/dashboard/:path*'],
};
```

<Callout type="info">
  For full-featured middleware with automatic refresh and RBAC checks, see [`@holeauth/nextjs-pages-router/middleware`](/packages/nextjs-pages-router#middleware).
</Callout>
</Step>

<Step>
### Wrap your app in HoleauthProvider

```tsx title="pages/_app.tsx"
import { HoleauthProvider } from '@holeauth/react';
import type { AppProps } from 'next/app';
import '@/styles/globals.css';

export default function App({ Component, pageProps }: AppProps) {
  return (
    <HoleauthProvider basePath="/api/auth" cookiePrefix="holeauth">
      <Component {...pageProps} />
    </HoleauthProvider>
  );
}
```

The provider makes session state available to all client components via hooks like [`useSession()`](/packages/react#useSession) and [`useAuth()`](/packages/react#useAuth).
</Step>

<Step>
### Create a protected page

Use `getServerSideProps` to read the session on the server:

```tsx title="pages/index.tsx"
import { GetServerSideProps } from 'next';
import Link from 'next/link';
import { getServerSidePropsSession } from '@holeauth/nextjs-pages-router';
import { auth } from '@/lib/auth';
import type { SessionData } from '@holeauth/core';

interface HomeProps {
  session: SessionData | null;
}

export default function HomePage({ session }: HomeProps) {
  if (!session) {
    return (
      <div>
        <h1>Welcome</h1>
        <p>You are not signed in.</p>
        <Link href="/login">Sign in</Link>
      </div>
    );
  }

  return (
    <div>
      <h1>Welcome back</h1>
      <p>User ID: {session.userId}</p>
      <p>Expires: {new Date(session.expiresAt * 1000).toLocaleString()}</p>
    </div>
  );
}

export const getServerSideProps: GetServerSideProps<HomeProps> = async (ctx) => {
  const session = await getServerSidePropsSession(ctx, auth.config);
  return { props: { session } };
};
```

For stricter guards with automatic redirect and RBAC checks, use [`withAuth`](/packages/nextjs-pages-router#withAuth):

```tsx title="pages/dashboard.tsx"
import { GetServerSideProps } from 'next';
import { withAuth } from '@holeauth/nextjs-pages-router';
import { auth } from '@/lib/auth';
import type { SessionData } from '@holeauth/core';

interface DashboardProps {
  userId: string;
  email: string;
}

export default function DashboardPage({ userId, email }: DashboardProps) {
  return (
    <div>
      <h1>Dashboard</h1>
      <p>Email: {email}</p>
      <p>User ID: {userId}</p>
    </div>
  );
}

export const getServerSideProps: GetServerSideProps<DashboardProps> = withAuth(
  async (ctx, session) => {
    const user = await auth.config.adapters.user.findById(session.userId);
    return {
      props: {
        userId: session.userId,
        email: user?.email ?? 'unknown',
      },
    };
  },
  auth.config,
  { redirectTo: '/login' },
);
```
