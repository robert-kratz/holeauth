# Role-based access control

This guide walks you through adding role-based access control to your Express application using `@holeauth/plugin-rbac`. You will define roles, groups, and permissions in a YAML file, then enforce them in your route handlers.

<Callout type="info">
  This guide assumes you have completed the [core Express setup](/getting-started/express).
</Callout>

### Extend your database schema

Import the RBAC table factory and add it to your schema:

<Tabs items={['Postgres', 'MySQL', 'SQLite']}>
  <Tab value="Postgres">
```ts title="db/schema.ts" {3,13-14}
import { pgTable, text, timestamp } from 'drizzle-orm/pg-core';
import { createHoleauthTables } from '@holeauth/adapter-drizzle/pg';
import { createRbacTables } from '@holeauth/rbac-drizzle/pg';

export const users = pgTable('user', {
  id: text('id').primaryKey().$defaultFn(() => crypto.randomUUID()),
  email: text('email').notNull().unique(),
  passwordHash: text('password_hash'),
  createdAt: timestamp('created_at', { withTimezone: true, mode: 'date' }).notNull().defaultNow(),
});

export const holeauth = createHoleauthTables({ usersTable: users });
export const rbac = createRbacTables({ usersTable: users });
```

<Step>
### Define roles in YAML

Create a YAML file defining your roles, groups, and permissions:

```yaml title="holeauth.rbac.yml"
groups:
  admin:
    name: Administrators
    description: Full system access
    permissions:
      - "*"
  
  editor:
    name: Editors
    description: Can create and edit content
    permissions:
      - "content:read"
      - "content:create"
      - "content:update"
      - "content:publish"
  
  viewer:
    name: Viewers
    description: Read-only access
    permissions:
      - "content:read"

roles:
  super_admin:
    name: Super Admin
    groups:
      - admin
  
  content_manager:
    name: Content Manager
    groups:
      - editor
    permissions:
      - "users:read"
  
  guest:
    name: Guest
    groups:
      - viewer
```

<Callout type="info">
  The YAML schema supports permission negation (`!permission:name`), group inheritance, and wildcard permissions (`resource:*` or `*`). See [`@holeauth/rbac-yaml`](/packages/plugin-rbac/yaml) for the full syntax.
</Callout>
</Step>

<Step>
### Wire the plugin into your auth instance

<Tabs items={['Postgres', 'MySQL', 'SQLite']}>
  <Tab value="Postgres">
```ts title="lib/auth.ts" {4-6,10,24-28}
import { createExpressAuth } from '@holeauth/express';
import { createHoleauthAdapters } from '@holeauth/adapter-drizzle/pg';
import { db } from '../db/client';
import { holeauth, rbac } from '../db/schema';
import { rbac as rbacPlugin } from '@holeauth/plugin-rbac';
import { createRbacAdapter } from '@holeauth/rbac-drizzle/pg';
import { loadRbacYaml } from '@holeauth/rbac-yaml';

const adapters = createHoleauthAdapters({ db, tables: holeauth.tables });
const rbacAdapter = createRbacAdapter({ db, tables: rbac.tables });

export const auth = createExpressAuth({
  secrets: {
    jwtSecret: process.env.HOLEAUTH_SECRET!,
  },
  adapters: {
    user: adapters.user,
    session: adapters.session,
    account: adapters.account,
    auditLog: adapters.auditLog,
    verificationToken: adapters.verificationToken,
    transaction: adapters.transaction,
  },
  plugins: [
    rbacPlugin({
      adapter: rbacAdapter,
      config: loadRbacYaml('./holeauth.rbac.yml'),
    }),
  ],
  tokens: {
    accessTtl: 60 * 15,
    refreshTtl: 60 * 60 * 24 * 30,
    cookiePrefix: 'holeauth',
    sameSite: 'lax',
  },
  registration: {
    selfServe: true,
    inviteTtlSeconds: 7 * 24 * 60 * 60,
    inviteUrl: ({ token }) =>
      `${process.env.APP_URL}/register/accept?token=${encodeURIComponent(token)}`,
  },
  allowDangerousEmailAccountLinking: false,
});
```

<Step>
### Enforce permissions in routes

Use the RBAC API to check permissions inside your route handlers:

```ts title="routes/content.ts"
import type { Request, Response } from 'express';
import { getSession } from '@holeauth/express';
import { auth } from '../lib/auth';

export async function createPost(req: Request, res: Response) {
  const session = await getSession(req, auth);
  
  if (!session) {
    return res.status(401).json({ error: 'Not authenticated' });
  }
  
  // Check if the user has the 'content:create' permission
  const canCreate = await auth.can(session.userId, 'content:create');
  
  if (!canCreate) {
    return res.status(403).json({ error: 'Forbidden: insufficient permissions' });
  }
  
  // Create the post...
  return res.json({ ok: true, postId: 'new-post-id' });
}

export async function deletePost(req: Request, res: Response) {
  const session = await getSession(req, auth);
  
  if (!session) {
    return res.status(401).json({ error: 'Not authenticated' });
  }
  
  // Check if the user has ALL of these permissions
  const canDelete = await auth.canAll(session.userId, ['content:delete', 'content:read']);
  
  if (!canDelete) {
    return res.status(403).json({ error: 'Forbidden: insufficient permissions' });
  }
  
  // Delete the post...
  return res.json({ ok: true });
}

export async function manageUsers(req: Request, res: Response) {
  const session = await getSession(req, auth);
  
  if (!session) {
    return res.status(401).json({ error: 'Not authenticated' });
  }
  
  // Check if the user has ANY of these permissions
  const canManage = await auth.canAny(session.userId, ['users:manage', '*']);
  
  if (!canManage) {
    return res.status(403).json({ error: 'Forbidden: insufficient permissions' });
  }
  
  // Manage users...
  return res.json({ ok: true });
}
```

<Callout type="info">
  The `auth` object returned by `createExpressAuth` includes the RBAC API when the plugin is registered. Use `auth.can(userId, permission)`, `auth.canAll(userId, permissions[])`, or `auth.canAny(userId, permissions[])` to check permissions.
</Callout>

## How it works

The RBAC plugin:

1. Loads the role and group definitions from YAML (or a plain object) at startup.
2. Stores user-role assignments in the database via the `RbacAdapter`.
3. At runtime, resolves a user effective permissions by:
   - Looking up the user assigned roles in the database.
   - Expanding each role into its groups and direct permissions.
   - Expanding each group into its permissions.
   - Applying negations (`!permission:name`).
   - Returning the final set of allowed permissions.
4. The `can` / `canAll` / `canAny` helpers match the requested permission(s) against the user effective permissions, supporting wildcards (`resource:*` or `*`).

See [@holeauth/plugin-rbac](/packages/plugin-rbac) for details on permission resolution order and wildcard matching.
