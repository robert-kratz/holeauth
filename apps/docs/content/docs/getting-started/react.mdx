# Getting started with @holeauth/react

`@holeauth/react` provides a context provider and typed hooks for accessing session state and triggering auth actions from client components.

<Tabs items={['pnpm', 'npm', 'yarn', 'bun']}>

```bash

```

<Steps>
<Step>

## Add the provider

```tsx title="app/layout.tsx"
import { HoleauthProvider } from '@holeauth/react';

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html>
      <body>
        <HoleauthProvider basePath="/api/auth">
          {children}
        </HoleauthProvider>
      </body>
    </html>
  );
}
```

</Step>
<Step>

## Use hooks in client components

```tsx title="components/user-menu.tsx"
'use client';
import { useSession, useSignOut } from '@holeauth/react';

export function UserMenu() {
  const { session, status } = useSession();
  const signOut = useSignOut();

  if (status === 'loading') return null;
  if (!session) return <a href="/login">Sign in</a>;

  return (
    <div>
      <span>{session.user.email}</span>
      <button onClick={() => signOut()}>Sign out</button>
    </div>
  );
}
```

</Step>
