# Passkeys

This guide walks you through adding WebAuthn-based passkey authentication to your Next.js App Router project. Users will register a passkey using their device biometrics or a hardware security key, then sign in without a password.

<Callout type="info">
  This guide assumes you have completed the [core Next.js App Router setup](/getting-started/nextjs-app-router).
</Callout>

import { pgTable, text, timestamp } from 'drizzle-orm/pg-core';
import { createHoleauthTables } from '@holeauth/adapter-drizzle/pg';
import { createPasskeyTables } from '@holeauth/passkey-drizzle/pg';

export const users = pgTable('user', {
  id: text('id').primaryKey().$defaultFn(() => crypto.randomUUID()),
  email: text('email').notNull().unique(),

<Step>
### Wire the plugin into your auth instance

<Tabs items={['Postgres', 'MySQL', 'SQLite']}>
  <Tab value="Postgres">
```ts title="lib/auth.ts" {4,5,9,23-29}
import { createAuthHandler } from '@holeauth/nextjs-app-router';
import { createHoleauthAdapters } from '@holeauth/adapter-drizzle/pg';
import { db } from '@/db/client';
import { holeauth, passkeys } from '@/db/schema';
import { passkey } from '@holeauth/plugin-passkey';
import { createPasskeyAdapter } from '@holeauth/passkey-drizzle/pg';

const adapters = createHoleauthAdapters({ db, tables: holeauth.tables });
const passkeyAdapter = createPasskeyAdapter({ db, tables: passkeys.tables });

export const auth = createAuthHandler({
  secrets: {
    jwtSecret: process.env.HOLEAUTH_SECRET!,
  },
  adapters: {
    user: adapters.user,
    session: adapters.session,
    account: adapters.account,
    auditLog: adapters.auditLog,
    verificationToken: adapters.verificationToken,
    transaction: adapters.transaction,
  },
  plugins: [
    passkey({
      adapter: passkeyAdapter,
      rpID: process.env.PASSKEY_RP_ID ?? 'localhost',
      rpName: 'MyApp',
      rpOrigin: process.env.APP_URL ?? 'http://localhost:3000',
    }),
  ],
  tokens: {
    accessTtl: 60 * 15,
    refreshTtl: 60 * 60 * 24 * 30,
    cookiePrefix: 'holeauth',
    sameSite: 'lax',
  },
  registration: {
    selfServe: true,
    inviteTtlSeconds: 7 * 24 * 60 * 60,
    inviteUrl: ({ token }) =>
      `${process.env.APP_URL}/register/accept?token=${encodeURIComponent(token)}`,
  },
  allowDangerousEmailAccountLinking: false,
});
```

<Step>
### Add the registration flow

Create server actions to initiate and complete passkey registration:

```ts title="app/actions/passkey.ts"
'use server';
import { auth } from '@/lib/auth';

export async function beginRegistration() {
  const session = await auth.getSession();
  if (!session) throw new Error('Unauthorized');

  const res = await fetch(`${process.env.APP_URL}/api/auth/passkey/register/begin`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ userId: session.userId }),
  });
  return res.json();
}

export async function completeRegistration(credential: unknown, deviceName?: string) {
  const session = await auth.getSession();
  if (!session) throw new Error('Unauthorized');

  const res = await fetch(`${process.env.APP_URL}/api/auth/passkey/register/complete`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ userId: session.userId, credential, deviceName }),
  });
  return res.json();
}
```

Build a client component that uses `@simplewebauthn/browser` to perform the WebAuthn ceremony:

```tsx title="app/settings/passkeys/register.tsx"
'use client';
import { useState } from 'react';
import { startRegistration } from '@simplewebauthn/browser';
import { beginRegistration, completeRegistration } from '@/app/actions/passkey';

export function RegisterPasskey() {
  const [deviceName, setDeviceName] = useState('');
  const [message, setMessage] = useState('');

  async function handleRegister() {
    try {
      const { options } = await beginRegistration();
      const credential = await startRegistration(options);
      await completeRegistration(credential, deviceName);
      setMessage('Passkey registered successfully');
    } catch (error) {
      setMessage(`Error: ${error instanceof Error ? error.message : String(error)}`);
    }
  }

  return (
    <div>
      <h2>Register a new passkey</h2>
      <input
        type="text"
        placeholder="Device name (optional)"
        value={deviceName}
        onChange={(e) => setDeviceName(e.target.value)}
      />
      <button onClick={handleRegister}>Register passkey</button>
      {message && <p>{message}</p>}
    </div>
  );
}
```

### Add the login flow

Create server actions for passkey authentication:

```ts title="app/actions/passkey.ts" {17-36}
'use server';
import { auth } from '@/lib/auth';

export async function beginRegistration() {
  const session = await auth.getSession();
  if (!session) throw new Error('Unauthorized');

  const res = await fetch(`${process.env.APP_URL}/api/auth/passkey/register/begin`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ userId: session.userId }),
  });
  return res.json();
}

export async function beginAuthentication(email: string) {
  const res = await fetch(`${process.env.APP_URL}/api/auth/passkey/login/begin`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ email }),
  });
  return res.json();
}

export async function completeAuthentication(
  pendingToken: string,
  credential: unknown,
  ip?: string,
  userAgent?: string,
) {
  const res = await fetch(`${process.env.APP_URL}/api/auth/passkey/login/complete`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ pendingToken, credential, ip, userAgent }),
  });
  return res.json();
}
```

Build a login component:

```tsx title="app/login/passkey.tsx"
'use client';
import { useState } from 'react';
import { useRouter } from 'next/navigation';
import { startAuthentication } from '@simplewebauthn/browser';
import { beginAuthentication, completeAuthentication } from '@/app/actions/passkey';

export function PasskeyLogin() {
  const [email, setEmail] = useState('');
  const [message, setMessage] = useState('');
  const router = useRouter();

  async function handleLogin() {
    try {
      const { pendingToken, options } = await beginAuthentication(email);
      const credential = await startAuthentication(options);
      await completeAuthentication(pendingToken, credential);
      router.push('/dashboard');
    } catch (error) {
      setMessage(`Error: ${error instanceof Error ? error.message : String(error)}`);
    }
  }

  return (
    <div>
      <h2>Sign in with passkey</h2>
      <input
        type="email"
        placeholder="Email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
      />
      <button onClick={handleLogin}>Sign in with passkey</button>
      {message && <p>{message}</p>}
    </div>
  );
}
```
</Step>

### Account management

Allow users to list and delete their passkeys:

```ts title="app/actions/passkey.ts" {38-55}
'use server';
import { auth } from '@/lib/auth';

// ... (beginRegistration, beginAuthentication, completeAuthentication)

export async function listPasskeys() {
  const session = await auth.getSession();
  if (!session) throw new Error('Unauthorized');

  const res = await fetch(`${process.env.APP_URL}/api/auth/passkey/list`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ userId: session.userId }),
  });
  return res.json();
}

export async function deletePasskey(credentialId: string) {
  const session = await auth.getSession();
  if (!session) throw new Error('Unauthorized');

  const res = await fetch(`${process.env.APP_URL}/api/auth/passkey/delete`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ userId: session.userId, credentialId }),
  });
  return res.json();
}
```

```tsx title="app/settings/passkeys/list.tsx"
'use client';
import { useEffect, useState } from 'react';
import { listPasskeys, deletePasskey } from '@/app/actions/passkey';

interface Passkey {
  id: string;
  credentialId: string;
  deviceName: string | null;
  createdAt: string;
}

export function PasskeyList() {
  const [passkeys, setPasskeys] = useState<Passkey[]>([]);

  useEffect(() => {
    listPasskeys().then(({ passkeys }) => setPasskeys(passkeys));
  }, []);

  async function handleDelete(credentialId: string) {
    await deletePasskey(credentialId);
    setPasskeys((prev) => prev.filter((p) => p.credentialId !== credentialId));
  }

  return (
    <div>
      <h2>Your passkeys</h2>
      {passkeys.length === 0 && <p>No passkeys registered.</p>}
      <ul>
        {passkeys.map((p) => (
          <li key={p.id}>
            {p.deviceName ?? 'Unnamed device'} — {new Date(p.createdAt).toLocaleDateString()}
            <button onClick={() => handleDelete(p.credentialId)}>Delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}
```

## How it works

1. **Registration**: The server generates a challenge and returns it as `options`. The browser calls `navigator.credentials.create()` with those options. The user approves with biometrics or a hardware key. The browser returns a signed credential. The server verifies the signature and stores the public key and credential ID.
2. **Authentication**: The server generates a challenge and returns it along with a list of credential IDs the user has registered. The browser calls `navigator.credentials.get()`. The authenticator signs the challenge. The server verifies the signature, matches the credential ID, and issues session tokens.

See [WebAuthn](https://webauthn.guide/) for a detailed explanation of the WebAuthn specification.

<Callout type="info">
  Passkey support varies by browser and platform. As of 2026, passkeys work on Chrome, Safari, Edge, and Firefox on desktop and mobile. Users on older browsers will need to fall back to password authentication.
</Callout>
