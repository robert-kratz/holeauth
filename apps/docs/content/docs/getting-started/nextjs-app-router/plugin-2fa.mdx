# Two-factor authentication

This guide walks you through adding TOTP-based two-factor authentication to your Next.js App Router project. Users will enroll by scanning a QR code with an authenticator app, then verify a six-digit code on every sign-in. Recovery codes are generated once during enrollment for account recovery.

<Callout type="info">
  This guide assumes you have completed the [core Next.js App Router setup](/getting-started/nextjs-app-router).
</Callout>

```ts title="db/schema.ts" {3,13-14}
import { pgTable, text, timestamp } from 'drizzle-orm/pg-core';
```

<Step>
### Wire the plugin into your auth instance

<Tabs items={['Postgres', 'MySQL', 'SQLite']}>
  <Tab value="Postgres">
```ts title="lib/auth.ts" {4,5,9,23-28}
import { createAuthHandler } from '@holeauth/nextjs-app-router';
import { createHoleauthAdapters } from '@holeauth/adapter-drizzle/pg';
import { db } from '@/db/client';
import { holeauth, twoFa } from '@/db/schema';
import { twofa } from '@holeauth/plugin-2fa';
import { createTwoFactorAdapter } from '@holeauth/2fa-drizzle/pg';

const adapters = createHoleauthAdapters({ db, tables: holeauth.tables });
const twoFactorAdapter = createTwoFactorAdapter({ db, tables: twoFa.tables });

export const auth = createAuthHandler({
  secrets: {
    jwtSecret: process.env.HOLEAUTH_SECRET!,
  },
  adapters: {
    user: adapters.user,
    session: adapters.session,
    account: adapters.account,
    auditLog: adapters.auditLog,
    verificationToken: adapters.verificationToken,
    transaction: adapters.transaction,
  },
  plugins: [
    twofa({
      adapter: twoFactorAdapter,
      issuer: 'MyApp',
      recoveryCodeCount: 10,
    }),
  ],
  tokens: {
    accessTtl: 60 * 15,
    refreshTtl: 60 * 60 * 24 * 30,
    cookiePrefix: 'holeauth',
    sameSite: 'lax',
  },
  registration: {
    selfServe: true,
    inviteTtlSeconds: 7 * 24 * 60 * 60,
    inviteUrl: ({ token }) =>
      `${process.env.APP_URL}/register/accept?token=${encodeURIComponent(token)}`,
  },
  allowDangerousEmailAccountLinking: false,
});
```


<Step>
### Add the enrollment flow

Create a server action to initiate enrollment:

```ts title="app/actions/twofa.ts"
'use server';
import { auth } from '@/lib/auth';

export async function setup2FA() {
  const session = await auth.getSession();
  if (!session) throw new Error('Unauthorized');

  const { secret, otpauthUrl, qrUrl } = await auth.twofa.setup(session.userId);
  return { secret, otpauthUrl, qrUrl };
}

export async function activate2FA(code: string) {
  const session = await auth.getSession();
  if (!session) throw new Error('Unauthorized');

  const { recoveryCodes } = await auth.twofa.activate(session.userId, code);
  return { recoveryCodes };
}
```

Build a client component to display the QR code and collect the verification code:

```tsx title="app/settings/2fa/enroll.tsx"
'use client';
import { useState } from 'react';
import { setup2FA, activate2FA } from '@/app/actions/twofa';

export function EnrollTwoFactor() {
  const [qrCode, setQrCode] = useState<string | null>(null);
  const [recoveryCodes, setRecoveryCodes] = useState<string[] | null>(null);
  const [code, setCode] = useState('');

  async function handleSetup() {
    const { qrUrl } = await setup2FA();
    setQrCode(qrUrl);
  }

  async function handleActivate() {
    const { recoveryCodes } = await activate2FA(code);
    setRecoveryCodes(recoveryCodes);
  }

  if (recoveryCodes) {
    return (
      <div>
        <h2>Two-factor authentication enabled</h2>
        <p>Save these recovery codes in a secure location. Each can be used once.</p>
        <pre>{recoveryCodes.join('\n')}</pre>
      </div>
    );
  }

  if (qrCode) {
    return (
      <div>
        <h2>Scan this QR code with your authenticator app</h2>
        <img src={qrCode} alt="2FA QR Code" />
        <input
          type="text"
          placeholder="Enter 6-digit code"
          value={code}
          onChange={(e) => setCode(e.target.value)}
        />
        <button onClick={handleActivate}>Verify and activate</button>
      </div>
    );
  }

  return <button onClick={handleSetup}>Enable two-factor authentication</button>;
}
```
</Step>

<Step>
### Handle the verification step at sign-in

When a user with 2FA enabled signs in, the plugin intercepts the flow and returns a `pending2fa` challenge. The `pendingToken` is written to the `holeauth.pending` HttpOnly cookie automatically. Use `onPending` to redirect straight to `/2fa/verify` — do **not** render `TwoFactorVerifyForm` inline inside the login page.

```tsx title="app/(guest)/login/page.tsx"
'use client';
import { useRouter } from 'next/navigation';
import { SignInForm } from '@holeauth/react-ui';

export default function LoginPage() {
  const router = useRouter();

  return (
    <SignInForm.Root
      onSuccess={() => router.push('/dashboard')}
      onPending={() => router.push('/2fa/verify')}
    >
      <SignInForm.Email />
      <SignInForm.Password />
      <SignInForm.Submit>Sign in</SignInForm.Submit>
      <SignInForm.Error />
    </SignInForm.Root>
  );
}
```

The `/2fa/verify` page uses `TwoFactorVerifyForm` to collect and submit the TOTP code. It reads the pending state from the `holeauth.pending` cookie — no token prop needed.

```tsx title="app/2fa/verify/page.tsx"
'use client';
import { useRouter } from 'next/navigation';
import { TwoFactorVerifyForm } from '@holeauth/react-ui';

export default function TwoFAVerifyPage() {
  const router = useRouter();

  return (
    <TwoFactorVerifyForm.Root
      onSuccess={() => router.push('/dashboard')}
      onCancel={() => router.push('/login')}
    >
      <TwoFactorVerifyForm.Code placeholder="000000" />
      <TwoFactorVerifyForm.Submit>Verify</TwoFactorVerifyForm.Submit>
      <TwoFactorVerifyForm.Error />
      <TwoFactorVerifyForm.Cancel>← Back to login</TwoFactorVerifyForm.Cancel>
    </TwoFactorVerifyForm.Root>
  );
}
```
</Step>

<Step>
### Recovery codes

<Callout type="warn">
  Recovery codes are single-use and stored hashed. Display them to the user exactly once during enrollment. Instruct users to save them in a password manager or print them.
</Callout>

Users can use a recovery code in place of a TOTP code when signing in. Each recovery code can be used exactly once. After consumption, it is removed from the database.
</Step>
</Steps>

## How it works

1. User signs in with email and password.
2. If 2FA is enabled, the plugin issues a `pendingToken` (short-lived JWT) and writes it to the `holeauth.pending` HttpOnly cookie. Session tokens are not issued yet.
3. The client collects a six-digit TOTP code (or a recovery code) from the user.
4. The client POSTs `{ code }` to `/api/auth/2fa/verify`. The `pendingToken` is read from the cookie server-side — no URL parameter or body field needed.
5. The plugin verifies the code, consumes the `pendingToken`, and issues full session tokens.
6. The browser receives the access and refresh cookies and the user is fully authenticated.

See [Sessions](/concepts/sessions) and [Token rotation](/concepts/token-rotation) for details on session lifecycle.
